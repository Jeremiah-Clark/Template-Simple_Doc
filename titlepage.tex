% titlepage.tex — Title page layout (pure LaTeX, edit per project)
%
% Available commands (populated from master.yaml via template.tex):
%   \doctitle           — Document title
%   \docauthor          — Author name
%   \docdate            — Date string
%   \docversion         — Version string (only defined if version: is set in YAML)
%   \doclogo            — Path to logo image (only defined if logo: is set in YAML)
%   \docdisclaimer      — Disclaimer text (only defined if disclaimer: is set in YAML)
%   \doclogowidth       — Logo width as a fraction of \textwidth (logo-width in YAML)
%   \doctppostrule      — Vertical space after the top rule, before the title
%   \doctpposttitle     — Vertical space after the title, before the author
%   \doctppostauthor    — Vertical space after the author, before the date
%   \doctppostdate      — Vertical space after the date, before the logo

\begin{titlepage}
  \thispagestyle{empty}

  % Version stamp — top-right corner
  \ifdefined\docversion
    \begin{flushright}
      \vspace*{-2.5em}
      {\small\sffamily\color{black!60} v\docversion}
    \end{flushright}
    \vspace{-2em}
  \fi

  \begin{center}
    \rule{\linewidth}{0.4pt}
    \vspace{\doctppostrule}

    {\Huge\sffamily \doctitle}

    \vspace{\doctpposttitle}
    {\large\sffamily \docauthor}

    \vspace{\doctppostauthor}
    {\docdate}

    \vspace{\doctppostdate}

    \ifdefined\doclogo
      \includegraphics[width=\doclogowidth\textwidth]{\doclogo}
    \fi

    \vfill

    \ifdefined\docdisclaimer
      \begin{important}
      \docdisclaimer
      \end{important}
    \fi

  \end{center}
\end{titlepage}
